### PERSON
  PKEY         INTEGER PRIMARY KEY
  NAME         TEXT NOT NULL
[INDEX] (NAME) UNIQUE

### DRUG
  PKEY         INTEGER PRIMARY KEY
  NAME         TEXT NOT NULL
[INDEX] (NAME) UNIQUE
  
### MEDICATION
  PKEY         INTEGER PRIMARY KEY
  PERSON_PKEY  INTEGER NOT NULL
  DRUG_PKEY    INTEGER NOT NULL
  DATE         INTEGER NOT NULL
[FOREIGN KEY] (PERSON_PKEY) => PERSON (PKEY)
[FOREIGN KEY] (DRUG_PKEY) => DRUG (PKEY)

### DRUG_COMPOSITION
  PKEY         INTEGER PRIMARY KEY
  NAME         TEXT NOT NULL
[INDEX] (NAME) UNIQUE

### DRUG_DOSAGE
  PKEY                     INTEGER PRIMARY KEY
  DRUG_PKEY                INTEGER NOT NULL
  DRUG_COMPOSITION_PKEY    INTEGER NOT NULL
  QUANTITY                 INTEGER NOT NULL
[FOREIGN KEY] (DRUG_PKEY) => DRUG (PKEY)
[FOREIGN KEY] (DRUG_COMPOSITION_PKEY) => DRUG_COMPOSITION (PKEY)
[INDEX] (DRUG_PKEY, DRUG_COMPOSITION_PKEY) UNIQUE
